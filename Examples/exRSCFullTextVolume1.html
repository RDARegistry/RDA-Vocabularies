@prefix ex: <http://example.com/> .
@prefix rdaa: <http://rdaregistry.info/Elements/a/> .
@prefix rdaco: <http://rdregistry.info/termList/RDAContentType/> .
@prefix rdact: <http://rdregistry.info/termList/RDACarrierType/> .
@prefix rdae: <http://rdaregistry.info/Elements/e/> .
@prefix rdam: <http://rdaregistry.info/Elements/m/> .
@prefix rdami: <http://rdregistry.info/termList/ModeIssue/> .
@prefix rdamt: <http://rdregistry.info/termList/RDAMediaType/> .
@prefix rdaw: <http://rdaregistry.info/Elements/w/> .
